--Contador 
library IEEE;
use IEEE.std_logic_1164.all;
use IEEE.numeric_std.all;

entity contador is
 	port (
 		reset	: in std_logic;
        clk 	: in std_logic;
        en		: in std_logic;
        val		: in unsigned(255 downto 0);
 		fin 	: out std_logic
 );
end contador;

architecture behavioural of contador is
	signal cnt_i : unsigned(val'range);
begin
 	counter: process (reset, clk)
 		begin
 			if reset = '1' then
            	cnt_i <= val;
            elsif rising_edge(clk) then
            	if en = '1' then
                	if cnt_i>0 then 
                    	fin <= '0';
 						cnt_i <= cnt_i - 1;
                    elsif cnt_i=0 then
                    	fin <='1';
                    end if;
                end if;
 			end if;
 	end process;
end behavioural;
